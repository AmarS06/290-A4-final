cmake_minimum_required(VERSION 3.20)
project(DataFrameLib LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

find_package(Arrow REQUIRED)
find_package(Parquet REQUIRED)

add_library(dataframelib
    src/column.cpp
    src/eager_dataframe.cpp
    src/expr.cpp
    src/lazy_dataframe.cpp
    src/optimizer.cpp
    src/plan_node.cpp
)

target_include_directories(dataframelib
    PUBLIC
        ${CMAKE_CURRENT_SOURCE_DIR}/include
)

target_link_libraries(dataframelib
    PUBLIC
        Arrow::arrow_shared
        Parquet::parquet_shared
)

add_executable(csv_smoketest tests/csv_smoketest.cpp)
target_link_libraries(csv_smoketest PRIVATE dataframelib)
target_compile_options(csv_smoketest PRIVATE -UNDEBUG)
target_compile_definitions(csv_smoketest PRIVATE
    SMOKETEST_DATA_DIR="${CMAKE_SOURCE_DIR}/data"
    SMOKETEST_GEN_DIR="${CMAKE_SOURCE_DIR}/generated")

install(TARGETS dataframelib
    ARCHIVE DESTINATION lib
    LIBRARY DESTINATION lib)
install(DIRECTORY include/ DESTINATION include)
